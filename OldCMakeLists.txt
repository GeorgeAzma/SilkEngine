cmake_minimum_required(VERSION 3.20)

project(VulkanEngine LANGUAGES CXX VERSION 1.0.0)

#additional cmake includes
include(FetchContent)
include(ExternalProject)

set(CMAKE_CXX_STANDARD 20)

if(NOT CMAKE_BUILD_TYPE)
	set(CMAKE_BUILD_TYPE Release)
endif()

if(MSVC)
	set(CMAKE_CXX_FLAGS_RELEASE "/O2")
    set(CMAKE_CXX_FLAGS "/bigobj /EHsc")
else()
	set(CMAKE_CXX_FLAGS_DEBUG "-g")
	set(CMAKE_CXX_FLAGS_RELEASE "-O3 -ffast-math")

	set(CMAKE_CXX_FLAGS "-lX11" /bigobj)
endif()

add_compile_definitions(GLEW_STATIC)
add_compile_definitions(_USE_MATH_DEFINES)

#vulkan
if (DEFINED VULKAN_SDK_PATH)
    list(APPEND INCLUDE_DIRS "${VULKAN_SDK_PATH}/Include")
    list(APPEND EXTRA_LIB_DIRS "${VULKAN_SDK_PATH}/Lib")
    set(Vulkan_FOUND "True")
else()
    find_package(Vulkan REQUIRED)
    message(STATUS "Found Vulkan: $ENV{VULKAN_SDK}")
endif()
if(NOT Vulkan_FOUND)
    message("Could not find vulkan")
else()
    message(STATUS "Using vulkan lib at: ${Vulkan_LIBRARIES}")
endif()

#Include sources

		main.cpp
        ${CPP_FILES}
)

list(APPEND EXTRA_LIBS Threads::Threads Vulkan::Vulkan)

#Create include directories
include_directories(
    ${PROJECT_SOURCE_DIR}/vendor
    ${PROJECT_SOURCE_DIR}/src
    ${INCLUDE_DIRS}
)

#Create exe
add_library(VulkanEngine SHARED ${SOURCE_FILES} ${files_imgui})

#Add dependencies
foreach(item IN LISTS DEPENDENCIES_TO_ADD)

    if("${${item}}" MATCHES TRUE)
        string(FIND ${item} "_" UNDERLINE_POS)
        string(SUBSTRING ${item} 0 ${UNDERLINE_POS} USABLE_STRING)
        string(TOUPPER ${USABLE_STRING} UPPER_STRING)
        add_dependencies(VulkanEngine ${UPPER_STRING})
    endif()

endforeach()

#Linking
target_link_libraries(VulkanEngine ${EXTRA_LIBS})
target_link_directories(VulkanEngine PUBLIC ${EXTRA_LIB_DIRS})

#Precompiling headers
target_precompile_headers(VulkanEngine PUBLIC "src/pch.h")

#Building shaders
find_program(GLSL_VALIDATOR glslangValidator HINTS 
  ${Vulkan_GLSLANG_VALIDATOR_EXECUTABLE} 
  /usr/bin 
  /usr/local/bin 
  ${VULKAN_SDK_PATH}/Bin
  ${VULKAN_SDK_PATH}/Bin32
  $ENV{VULKAN_SDK}/Bin/ 
  $ENV{VULKAN_SDK}/Bin32/
)
 
file(GLOB_RECURSE GLSL_SOURCE_FILES
    "${PROJECT_SOURCE_DIR}/res/shaders/*.vert"
    "${PROJECT_SOURCE_DIR}/res/shaders/*.frag"
)
 
foreach(GLSL ${GLSL_SOURCE_FILES})
    get_filename_component(FILE_NAME ${GLSL} NAME)
    set(SPIRV "${PROJECT_BINARY_DIR}/res/shaders/build/${FILE_NAME}.spv")
    add_custom_command(
        OUTPUT ${SPIRV}
        COMMAND ${CMAKE_COMMAND} -E make_directory "${PROJECT_BINARY_DIR}/res/shaders/build"
        COMMAND ${GLSL_VALIDATOR} -V ${GLSL} -o ${SPIRV}
        DEPENDS ${GLSL})
    list(APPEND SPIRV_BINARY_FILES ${SPIRV})
endforeach(GLSL)
 
add_custom_target(
    Shaders
    DEPENDS ${SPIRV_BINARY_FILES}
)

add_dependencies(VulkanEngine Shaders)

#Copy res folder where exe file is
add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
		COMMAND ${CMAKE_COMMAND} -E copy_directory ${CMAKE_CURRENT_SOURCE_DIR}/res  $<TARGET_FILE_DIR:VulkanEngine>/res
		)

add_custom_command(TARGET VulkanEngine POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E make_directory "$<TARGET_FILE_DIR:VulkanEngine>/res/shaders/build"
    COMMAND ${CMAKE_COMMAND} -E copy_directory
        "${PROJECT_BINARY_DIR}/res"
        "$<TARGET_FILE_DIR:VulkanEngine>/res"
        )
